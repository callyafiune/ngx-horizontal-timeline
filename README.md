# Installation

```javascript
$ npm i ngx-horizontal-timeline
```
## Import

```javascript
import { NgxHorizontalTimelineModule } from 'ngx-horizontal-timeline';
@NgModule({
  declarations: [
    AppComponent
  ],
  imports: [
    NgxHorizontalTimelineModule
    ...
  ],
  providers: [],
  bootstrap: [AppComponent]
})
export class AppModule { }
```
## Component

```javascript
...
items: TimelineItem[];
 
ngOnInit() {
  this.items.push({
    label: 'Test 1',
    icon: 'glyphicon glyphicon-adjust',
    active: true,
    title: 'Example 1',

```
## HTML

